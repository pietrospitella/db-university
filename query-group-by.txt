/* 1. Contare quanti iscritti ci sono stati ogni anno */

SELECT COUNT(YEAR(`students`.`enrolment_date`)) AS `numero_iscritti`, YEAR(`students`.`enrolment_date`) AS `anno_iscrizione` 
FROM `students` 
GROUP BY `anno_iscrizione`;

/* 2. Contare gli insegnanti che hanno l'ufficio nello stesso edificio */ 

SELECT COUNT(`teachers`.`id`) AS `numero_insegnanti`, `office_address` AS `nome_ufficio` 
FROM `teachers` 
GROUP BY `nome_ufficio`;

/* 3. Calcolare la media dei voti di ogni appello d'esame */

SELECT AVG(`vote`), `exam_id` 
FROM `exam_student` 
GROUP BY `exam_id`;

/* 4. Contare quanti corsi di laurea ci sono per ogni dipartimento */

SELECT COUNT(id), `department_id`
FROM `degrees`
GROUP BY `department_id`;
